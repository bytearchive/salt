   45 -- | Randomly generate a public and private key for doing
   46 -- authenticated signing and verification.
   47 createKeypair :: IO KeyPair
   48 createKeypair = do
   49   pk <- SI.mallocByteString signPublicKeyLength
   50   sk <- SI.mallocByteString signSecretKeyLength
   51 
   52   void $ withForeignPtr pk $ \ppk ->
   53     void $ withForeignPtr sk $ \psk ->
   54       c_crypto_sign_keypair ppk psk
   55       
   56   return (PublicKey $ SI.fromForeignPtr pk 0 signPublicKeyLength,
   57           SecretKey $ SI.fromForeignPtr sk 0 signSecretKeyLength)
   58 
   59 -- | Sign a message with a particular 'SecretKey'.
   60 sign :: SecretKey 
   61      -- ^ Signers secret key
   62      -> ByteString 
   63      -- ^ Input message
   64      -> ByteString
   65      -- ^ Resulting signed message
   66 sign (SecretKey sk) xs =
   67   unsafePerformIO $ SU.unsafeUseAsCStringLen xs $ \(mstr,mlen) ->
   68     SU.unsafeUseAsCString sk $ \psk ->
   69       SI.createAndTrim (mlen+sign_BYTES) $ \out ->
   70        fromIntegral `liftM` c_crypto_sign out mstr (fromIntegral mlen) psk
   71 {-# INLINEABLE sign #-}
   72 
   73 -- | Verifies a signed message with a 'PublicKey'. Returns @Nothing@ if
   74 -- verification fails, or @Just xs@ where @xs@ is the original message if it
   75 -- succeeds.
   76 verify :: PublicKey
   77        -- ^ Signers public key
   78        -> ByteString
   79        -- ^ Signed message
   80        -> Maybe ByteString
   81        -- ^ Verification check
   82 verify (PublicKey pk) xs =
   83   unsafePerformIO $ SU.unsafeUseAsCStringLen xs $ \(smstr,smlen) ->
   84     SU.unsafeUseAsCString pk $ \ppk ->
   85       alloca $ \pmlen -> do
   86         out <- SI.mallocByteString smlen
   87         
   88         r <- withForeignPtr out $ \pout -> 
   89                c_crypto_sign_open pout pmlen smstr (fromIntegral smlen) ppk
   90         
   91         if r /= 0 then return Nothing
   92           else do
   93             l <- peek pmlen
   94             return $ Just $ SI.fromForeignPtr out 0 (fromIntegral l)
   95 {-# INLINEABLE verify #-}
   96

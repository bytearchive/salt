   53 newtype PKNonce = PKNonce ByteString deriving (Show, Eq)
   54 instance Nonce PKNonce where
   55   {-# SPECIALIZE instance Nonce PKNonce #-}
   56   size = Tagged nonceLength
   57   toBS (PKNonce b)   = b
   58   fromBS x
   59     | S.length x == nonceLength = Just (PKNonce x)
   60     | otherwise                 = Nothing

   66 -- | Randomly generate a public and private key
   67 -- for doing authenticated encryption.
   68 createKeypair :: IO KeyPair
   69 createKeypair = do
   70   pk <- SI.mallocByteString publicKeyLength
   71   sk <- SI.mallocByteString secretKeyLength
   72 
   73   void $ withForeignPtr pk $ \ppk ->
   74     void $ withForeignPtr sk $ \psk ->
   75       c_crypto_box_keypair ppk psk
   76       
   77   return (PublicKey $ SI.fromForeignPtr pk 0 publicKeyLength, 
   78           SecretKey $ SI.fromForeignPtr sk 0 secretKeyLength)

   80 encrypt :: PKNonce
   81         -- ^ Nonce
   82         -> ByteString
   83         -- ^ Message
   84         -> PublicKey
   85         -- ^ Recievers public key
   86         -> SecretKey
   87         -- ^ Senders secret key
   88         -> ByteString 
   89         -- ^ Ciphertext
   90 encrypt (PKNonce n) msg (PublicKey pk) (SecretKey sk) = unsafePerformIO $ do
   91   (c, m) <- prepEnc msg
   92   let mlen = S.length m
   93   
   94   -- as you can tell, this is unsafe
   95   void $ withForeignPtr c $ \pc ->
   96     SU.unsafeUseAsCString m $ \pm ->
   97       SU.unsafeUseAsCString n $ \pn -> 
   98         SU.unsafeUseAsCString pk $ \ppk ->
   99           SU.unsafeUseAsCString sk $ \psk ->
  100             c_crypto_box pc pm (fromIntegral mlen) pn ppk psk
  101   
  102   let r = SI.fromForeignPtr c 0 mlen
  103   return $ SU.unsafeDrop msg_BOXZEROBYTES r
  104 {-# INLINEABLE encrypt #-}
  105   
  106 decrypt :: PKNonce
  107         -- ^ Nonce
  108         -> ByteString
  109         -- ^ Input ciphertext
  110         -> PublicKey
  111         -- ^ Senders public key
  112         -> SecretKey
  113         -- ^ Recievers secret key
  114         -> Maybe ByteString -- ^ Ciphertext
  115 decrypt (PKNonce n) cipher (PublicKey pk) (SecretKey sk) = unsafePerformIO $ do
  116   (m, c) <- prepDec cipher
  117   let clen = S.length c
  118   
  119   -- as you can tell, this is unsafe
  120   r <- withForeignPtr m $ \pm ->
  121     SU.unsafeUseAsCString c $ \pc ->
  122       SU.unsafeUseAsCString n $ \pn -> 
  123         SU.unsafeUseAsCString pk $ \ppk ->
  124           SU.unsafeUseAsCString sk $ \psk ->
  125             c_crypto_box_open pm pc (fromIntegral clen) pn ppk psk
  126 
  127   return $ if r /= 0 then Nothing
  128             else
  129              let bs = SI.fromForeignPtr m 0 clen
  130              in Just $ SU.unsafeDrop msg_ZEROBYTES bs
  131 {-# INLINEABLE decrypt #-}

  134 -- 
  135 -- Precomputation interface
  136 -- 
  137 
  138 {- $precomp
  139 
  140 If you send many messages to the same receiver, or receive many
  141 messages from the same sender, you can gain speed increases by instead
  142 using the following precomputation interface, which splits the
  143 encryption and decryption steps into two parts.
  144 
  145 For encryption, you first create an 'NM' by using 'createNM', using
  146 the senders secret key, and receivers public key. You can then use
  147 'encryptNM' to encrypt data.
  148 
  149 For decryption, you first create an 'NM' by using 'createNM', using
  150 the recievers secret key, and the senders publickey. You can then use
  151 'decryptNM' to decrypt data.
  152 
  153 -}
  154 
  155 -- | An 'NM' is intermediate data computed by 'createNM' given a
  156 -- public and private key which can be used to encrypt/decrypt
  157 -- information via 'encryptNM' or 'decryptNM'.
  158 -- 
  159 -- An 'NM' can be re-used between two communicators for any number of
  160 -- messages.
  161 -- 
  162 -- Its name is not particularly enlightening as to its purpose, it is merely the same
  163 -- identifier used in the NaCl source code for this interface.
  164 newtype NM = NM ByteString deriving (Eq, Show)
  165 
  166 -- | Creates an intermediate piece of 'NM' data for sending/receiving
  167 -- messages to/from the same person. The resulting 'NM' can be used for
  168 -- any number of messages between client/server.
  169 createNM :: KeyPair -> NM
  170 createNM (PublicKey pk, SecretKey sk) = unsafePerformIO $ do
  171   nm <- SI.mallocByteString nmLength
  172   void $ withForeignPtr nm $ \pnm ->
  173     SU.unsafeUseAsCString pk $ \ppk ->
  174       SU.unsafeUseAsCString sk $ \psk ->
  175         c_crypto_box_beforenm pnm ppk psk
  176   return $ NM $ SI.fromForeignPtr nm 0 nmLength
  177 {-# INLINEABLE createNM #-}
  178 
  179 -- | Encrypt data from a specific sender to a specific receiver with
  180 -- some precomputed 'NM' data.
  181 encryptNM :: NM -> PKNonce -> ByteString -> ByteString
  182 encryptNM (NM nm) (PKNonce n) msg = unsafePerformIO $ do
  183   (c, m) <- prepEnc msg
  184   let mlen = S.length m
  185 
  186   -- as you can tell, this is unsafe
  187   void $ withForeignPtr c $ \pc ->
  188     SU.unsafeUseAsCString m $ \pm ->
  189       SU.unsafeUseAsCString n $ \pn -> 
  190         SU.unsafeUseAsCString nm $ \pnm ->
  191           c_crypto_box_afternm pc pm (fromIntegral mlen) pn pnm
  192   
  193   let r = SI.fromForeignPtr c 0 mlen
  194   return $ SU.unsafeDrop msg_BOXZEROBYTES r
  195 {-# INLINEABLE encryptNM #-}
  196 
  197 -- | Decrypt data from a specific sender for a specific receiver with
  198 -- some precomputed 'NM' data.
  199 decryptNM :: NM -> PKNonce -> ByteString -> Maybe ByteString
  200 decryptNM (NM nm) n cipher = unsafePerformIO $ do
  201   (m, c) <- prepDec cipher
  202   let clen = S.length c
  203   
  204   -- as you can tell, this is unsafe
  205   r <- withForeignPtr m $ \pm ->
  206     SU.unsafeUseAsCString c $ \pc ->
  207       SU.unsafeUseAsCString (toBS n) $ \pn -> 
  208         SU.unsafeUseAsCString nm $ \pnm ->
  209           c_crypto_box_open_afternm pm pc (fromIntegral clen) pn pnm
  210 
  211   return $ if r /= 0 then Nothing
  212             else
  213              let bs = SI.fromForeignPtr m 0 clen
  214              in Just $ SU.unsafeDrop msg_ZEROBYTES bs
  215 {-# INLINEABLE decryptNM #-}
  216 
  217 -- 
  218 -- Inputs to encrypt/decrypt must be padded
  219 -- 
  220 
  221 prepEnc :: ByteString -> IO (ForeignPtr Word8, ByteString)
  222 prepEnc bs = do
  223   c <- SI.mallocByteString (S.length bs + msg_ZEROBYTES)
  224   return $! (c, S.replicate msg_ZEROBYTES 0x0 `S.append` bs)
  225   
  226 prepDec :: ByteString -> IO (ForeignPtr Word8, ByteString)
  227 prepDec bs = do
  228   c <- SI.mallocByteString (S.length bs + msg_BOXZEROBYTES)
  229   return $! (c, S.replicate msg_BOXZEROBYTES 0x0 `S.append` bs)

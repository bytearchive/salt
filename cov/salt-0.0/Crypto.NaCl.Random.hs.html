   10 -- Obtaining random bytes via @\/dev\/urandom@. Useful for nonces or similar.
   11 -- 
   12 -- A package like @mwc-random@ would also work for getting at
   13 -- randomness.  This is really only here for completeness because
   14 -- internally certain NaCl primitives use the @randombytes@ call.
   15 -- 
   16 
   17 module Crypto.NaCl.Random
   18        ( randomBytes -- :: Int -> IO ByteString
   19        ) where
   20 import Foreign.C.Types
   21 import Foreign.Ptr
   22 import Control.Monad (void)
   23 import Data.Word
   24 
   25 import Data.ByteString as S
   26 import Data.ByteString.Internal as SI
   27 
   28 -- | Generate a random ByteString which is internally based on @\/dev\/urandom@.
   29 randomBytes :: Int -> IO ByteString
   30 randomBytes n 
   31   | n < 0     = error "Crypto.NaCl.Random.randomBytes: length must be greater than 0"
   32   | otherwise = SI.create n $ \out -> void $ c_randombytes out (fromIntegral n)
   33 
   34 --
   35 -- FFI
   36 -- 
   37 
   38 foreign import ccall unsafe "randombytes"
   39   c_randombytes :: Ptr Word8 -> CULLong -> IO Int

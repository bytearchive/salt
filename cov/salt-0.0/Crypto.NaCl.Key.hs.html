   10 -- This module exports two simple types, 'PublicKey' and 'SecretKey'
   11 -- which are merely @newtype@s for 'ByteString'. This is for type safety
   12 -- so you never confuse your keys or accidentally give them 'ByteString's
   13 -- when you shouldn't have.
   14 module Crypto.NaCl.Key
   15        ( -- * Types
   16          PublicKey(..)
   17        , SecretKey(..)
   18        , KeyPair
   19        ) where
   20 import Data.ByteString
   21 
   22 newtype PublicKey = PublicKey { unPublicKey :: ByteString }
   23         deriving (Eq, Show)
   24 
   25 newtype SecretKey = SecretKey { unSecretKey :: ByteString }
   26         deriving (Eq, Show)
   27 
   28 type KeyPair = (PublicKey, SecretKey)

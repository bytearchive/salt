   13 module Crypto.NaCl.Nonce
   14   ( -- * Nonce class
   15     Nonce, size, toBS, fromBS
   16   , createZeroNonce   -- :: Nonce k => k
   17   , createRandomNonce -- :: Nonce k => IO k
   18   , clearBytes        -- :: Nonce k => Int -> k -> k
   19   , incNonce          -- :: Nonce k => k -> k
   20   ) where
   21 import Foreign.Ptr
   22 import Foreign.C.Types
   23 import Control.Monad (liftM)
   24 import Data.Word
   25 import Data.Maybe
   26 import Data.Tagged
   27 import Data.ByteString as S
   28 import Data.ByteString.Internal as SI
   29 import Data.ByteString.Unsafe as SU
   30 
   31 import Crypto.NaCl.Random (randomBytes)
   32 
   33 class Nonce k where
   34   size              :: Tagged k Int
   35   toBS              :: k -> ByteString
   36   fromBS            :: ByteString -> (Maybe k)
   37 
   38   -- | Create an empty nonce of length @n@ where all the bytes are zero.
   39   createZeroNonce :: k
   40   createZeroNonce
   41     = fromJust . fromBS $ S.replicate sz 0x0
   42     where sz = unTagged (size :: Tagged k Int)
   43   {-# INLINEABLE createZeroNonce #-}
   44 
   45   -- | Create a random nonce, seeded by @/dev/urandom@.
   46   createRandomNonce :: IO k
   47   createRandomNonce
   48     = (fromJust . fromBS) `liftM` randomBytes sz
   49     where sz = unTagged (size :: Tagged k Int)
   50   {-# INLINEABLE createRandomNonce #-}
   51   
   52   -- | @clearBytes n nonce@ clears the last @n@ bytes of the Nonce and
   53   -- makes them all 0. This is useful for the pattern of generating a
   54   -- cryptographic nonce randomly, clearing the last @n@ bytes, and then
   55   -- using 'incNonce' to increment the 'Nonce' for communication with
   56   -- another party.
   57   -- 
   58   -- Invariants:
   59   -- 
   60   -- * @n@ must be less than the size of the @nonce@
   61   -- 
   62   -- Properties:
   63   -- 
   64   -- > clearBytes (nonceLen nonce) nonce == createZeroNonce (nonceLen nonce)
   65   -- 
   66   clearBytes :: Int -> k -> k
   67   clearBytes n x
   68     | n > l  = error "Crypto.NaCl.Nonce.clearBytes: n > length of nonce"
   69     | n < 0  = error "Crypto.NaCl.Nonce.clearBytes: n < 0"  
   70     | n == 0 = x
   71     | otherwise = (fromJust . fromBS) $! S.take (l - n) (toBS x) `S.append` S.replicate n 0x0
   72     where l = unTagged (size :: Tagged k Int)
   73   {-# INLINEABLE clearBytes #-}
   74 
   75   -- | Increment a Nonce by 1.
   76   incNonce :: k -> k
   77   incNonce n =
   78     (fromJust . fromBS) $ SI.unsafeCreate l $ \out -> do
   79       SU.unsafeUseAsCStringLen (toBS n) $ \(b,blen) ->
   80         SI.memcpy out (castPtr b) (fromIntegral blen)
   81       c_incnonce out (fromIntegral l)
   82     where l = unTagged (size :: Tagged k Int)
   83   {-# INLINEABLE incNonce #-}
   84

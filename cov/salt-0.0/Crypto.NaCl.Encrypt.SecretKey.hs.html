   41 data SKNonce = SKNonce ByteString deriving (Show, Eq)
   42 instance Nonce SKNonce where
   43   {-# SPECIALIZE instance Nonce SKNonce #-}
   44   size = Tagged nonceLength
   45   toBS (SKNonce b)   = b
   46   fromBS x
   47     | S.length x == nonceLength = Just (SKNonce x)
   48     | otherwise                 = Nothing

   51 encrypt :: SKNonce
   52         -- ^ Nonce
   53         -> ByteString
   54         -- ^ Input
   55         -> SecretKey
   56         -- ^ Secret key
   57         -> ByteString
   58         -- ^ Ciphertext
   59 encrypt (SKNonce n) msg (SecretKey k) = unsafePerformIO $ do
   60   let mlen = S.length msg + msg_ZEROBYTES
   61   c <- SI.mallocByteString mlen
   62   
   63   -- inputs to crypto_box must be padded
   64   let m = S.replicate msg_ZEROBYTES 0x0 `S.append` msg
   65   
   66   -- as you can tell, this is unsafe
   67   void $ withForeignPtr c $ \pc ->
   68     SU.unsafeUseAsCString m $ \pm ->
   69       SU.unsafeUseAsCString n $ \pn -> 
   70         SU.unsafeUseAsCString k $ \pk ->
   71           c_crypto_secretbox pc pm (fromIntegral mlen) pn pk
   72   
   73   let r = SI.fromForeignPtr c 0 mlen
   74   return $ SU.unsafeDrop msg_BOXZEROBYTES r
   75 {-# INLINEABLE encrypt #-}
   76 
   77 decrypt :: SKNonce
   78         -- ^ Nonce
   79         -> ByteString
   80         -- ^ Input
   81         -> SecretKey        
   82         -- ^ Secret key
   83         -> Maybe ByteString 
   84         -- ^ Ciphertext
   85 decrypt (SKNonce n) cipher (SecretKey k) = unsafePerformIO $ do
   86   let clen = S.length cipher + msg_BOXZEROBYTES
   87   m <- SI.mallocByteString clen
   88   
   89   -- inputs to crypto_box must be padded
   90   let c = S.replicate msg_BOXZEROBYTES 0x0 `S.append` cipher
   91   
   92   -- as you can tell, this is unsafe
   93   r <- withForeignPtr m $ \pm ->
   94     SU.unsafeUseAsCString c $ \pc ->
   95       SU.unsafeUseAsCString n $ \pn -> 
   96         SU.unsafeUseAsCString k $ \pk ->
   97           c_crypto_secretbox_open pm pc (fromIntegral clen) pn pk
   98   
   99   return $ if r /= 0 then Nothing
  100             else
  101              let bs = SI.fromForeignPtr m 0 clen
  102              in Just $ SU.unsafeDrop msg_ZEROBYTES bs
  103 {-# INLINEABLE decrypt #-}

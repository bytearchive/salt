   42 data StreamNonce = StreamNonce ByteString deriving (Show, Eq)
   43 instance Nonce StreamNonce where
   44   {-# SPECIALIZE instance Nonce StreamNonce #-}
   45   size = Tagged nonceLength
   46   toBS (StreamNonce b)   = b
   47   fromBS x
   48     | S.length x == nonceLength = Just (StreamNonce x)
   49     | otherwise                 = Nothing
   50 
   51 -- | Given a 'Nonce' @n@, size @s@ and 'SecretKey' @sk@, @streamGen n
   52 -- s sk@ generates a cryptographic stream of length @s@.
   53 streamGen :: StreamNonce
   54           -- ^ Nonce
   55           -> Int
   56           -- ^ Size
   57           -> SecretKey
   58           -- ^ Input
   59           -> ByteString
   60           -- ^ Resulting crypto stream
   61 streamGen (StreamNonce n) sz (SecretKey sk)
   62   | S.length sk /= keyLength
   63   = error "Crypto.NaCl.Encrypt.Stream.XSalsa20.streamGen: bad key length"
   64   | otherwise
   65   = unsafePerformIO . SI.create sz $ \out ->
   66     SU.unsafeUseAsCString n $ \pn ->
   67       SU.unsafeUseAsCString sk $ \psk ->
   68         void $ c_crypto_stream_xsalsa20 out (fromIntegral sz) pn psk
   69 {-# INLINEABLE streamGen #-}
   70 
   71 -- | Given a 'Nonce' @n@, plaintext @p@ and 'SecretKey' @sk@, @encrypt n p sk@ encrypts the message @p@ using 'SecretKey' @sk@ and returns the result.
   72 -- 
   73 -- 'encrypt' guarantees the resulting ciphertext is the plaintext
   74 -- bitwise XOR'd with the result of 'streamGen'. As a result,
   75 -- 'encrypt' can also be used to decrypt messages.
   76 encrypt :: StreamNonce
   77         -- ^ Nonce
   78         -> ByteString
   79         -- ^ Input plaintext
   80         -> SecretKey
   81         -- ^ Secret key
   82         -> ByteString
   83         -- ^ Ciphertext
   84 encrypt (StreamNonce n) msg (SecretKey sk)
   85   | S.length sk /= keyLength
   86   = error "Crypto.NaCl.Encrypt.Stream.XSalsa20.encrypt: bad key length"
   87   | otherwise
   88   = let l = S.length msg
   89     in unsafePerformIO . SI.create l $ \out ->
   90       SU.unsafeUseAsCString msg $ \cstr -> 
   91         SU.unsafeUseAsCString n $ \pn ->
   92           SU.unsafeUseAsCString sk $ \psk ->
   93             void $ c_crypto_stream_xsalsa20_xor out cstr (fromIntegral l) pn psk
   94 {-# INLINEABLE encrypt #-}
   95 
   96 -- | Simple alias for 'encrypt'.
   97 decrypt :: StreamNonce
   98         -- ^ Nonce
   99         -> ByteString
  100         -- ^ Input ciphertext
  101         -> SecretKey
  102         -- ^ Secret key
  103         -> ByteString
  104         -- ^ Plaintext
  105 decrypt n c sk = encrypt n c sk
  106 {-# INLINEABLE decrypt #-}

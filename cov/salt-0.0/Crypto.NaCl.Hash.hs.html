   10 -- Fast, cryptographically strong hashing functions. Currently
   11 -- only provides SHA-512 and SHA-256.
   12 -- 
   13 module Crypto.NaCl.Hash
   14        ( sha512 -- :: ByteString -> ByteString
   15        , sha256 -- :: ByteString -> ByteString
   16        ) where
   17 import Foreign.C
   18 import Foreign.Ptr
   19 import Control.Monad (void)
   20 import Data.Word
   21 import System.IO.Unsafe (unsafePerformIO)
   22 
   23 import Data.ByteString as S
   24 import Data.ByteString.Internal as SI
   25 import Data.ByteString.Unsafe as SU
   26 
   27 sha512 :: ByteString -> ByteString
   28 sha512 xs =
   29   -- The default primitive of SHA512 has 64 bytes of output.      
   30   unsafePerformIO . SI.create 64 $ \out ->
   31     SU.unsafeUseAsCStringLen xs $ \(cstr,clen) ->
   32       void $ c_crypto_hash_sha512 out cstr (fromIntegral clen)
   33 {-# INLINEABLE sha512 #-}
   34 
   35 sha256 :: ByteString -> ByteString
   36 sha256 xs =
   37   -- SHA256 has 32 bytes of output
   38   unsafePerformIO . SI.create 32 $ \out ->
   39     SU.unsafeUseAsCStringLen xs $ \(cstr,clen) ->
   40       void $ c_crypto_hash_sha256 out cstr (fromIntegral clen)
   41 {-# INLINEABLE sha256 #-}
   42 
   43 --
   44 -- FFI
   45 --
   46 
   47 type HashFunc = Ptr Word8 -> Ptr CChar -> CULLong -> IO CInt
   48 
   49 foreign import ccall unsafe "glue_crypto_hash_sha512"
   50   c_crypto_hash_sha512 :: HashFunc
   51 
   52 foreign import ccall unsafe "glue_crypto_hash_sha256"
   53   c_crypto_hash_sha256 :: HashFunc

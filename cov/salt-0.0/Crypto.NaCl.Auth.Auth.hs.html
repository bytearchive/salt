   33 authenticate :: SecretKey
   34              -- ^ Secret key
   35              -> ByteString
   36              -- ^ Message 
   37              -> ByteString
   38              -- ^ Authenticator
   39 authenticate (SecretKey k) msg = 
   40   unsafePerformIO . SI.create auth_BYTES $ \out ->
   41     SU.unsafeUseAsCStringLen msg $ \(cstr, clen) ->
   42       SU.unsafeUseAsCString k $ \pk ->
   43         void $ c_crypto_auth out cstr (fromIntegral clen) pk
   44 {-# INLINEABLE authenticate #-}
   45 
   46 verify :: SecretKey
   47        -- ^ Key
   48        -> ByteString 
   49        -- ^ Authenticator returned via 'authenticate'
   50        -> ByteString 
   51        -- ^ Message
   52        -> Bool
   53        -- ^ Result: @True@ if properly verified, @False@ otherwise
   54 verify (SecretKey k) auth msg =
   55   unsafePerformIO $ SU.unsafeUseAsCString auth $ \pauth ->
   56     SU.unsafeUseAsCStringLen msg $ \(cstr, clen) ->
   57       SU.unsafeUseAsCString k $ \pk -> do
   58         b <- c_crypto_auth_verify pauth cstr (fromIntegral clen) pk
   59         return (b == 0)
   60 {-# INLINEABLE verify #-}
   61

   66 -- | @authKeyLength@ is the required key length for a key given
   67 -- to 'authenticate' or 'verify'. Using any other key length will
   68 -- result in error.
   69 authKeyLength :: Int
   70 authKeyLength = #{const crypto_auth_KEYBYTES}

   72 auth_BYTES :: Int
   73 auth_BYTES = #{const crypto_auth_BYTES}
